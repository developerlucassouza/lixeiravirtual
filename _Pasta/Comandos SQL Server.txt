-Criar Trigger
CREATE TRIGGER tgrVendaInserida
ON tab_vendas
AS
BEGIN
   DECLARE
   @var_valor DECIMAL(10, 2)
   @var_date DATE
   SELECT @var_date = data, @var_valor = valor FROM INSERTED
   UPDATE tab_caixa SET saldo_final = saldo_final + @var_valor
   WHERE data = @var_date

-Criar Backup
